# Discuss correlation
correlation_math_reading = df['math score'].corr(df['reading score'])
print(f"\nCorrelation between Math Score and Reading Score: {correlation_math_reading:.2f}")
# Discussion of correlation
if correlation_math_reading > 0:
    correlation_description = "There is a positive correlation; generally, as math scores increase, reading scores also tend to increase."
elif correlation_math_reading < 0:
    correlation_description = "There is a negative correlation; as math scores increase, reading scores tend to decrease."
else:
    correlation_description = "There is no significant correlation between math scores and reading scores."

print(correlation_description)
